94template <typename ExecSpace, typename SourceType, typename DestinationType>
95void convert(
96 const SourceType& src, DestinationType& dst,
97 typename std::enable_if<
98 Morpheus::is_coo_matrix_format_container_v<SourceType> &&
99 Morpheus::is_csr_matrix_format_container_v<DestinationType> &&
100 !Morpheus::is_generic_space_v<ExecSpace> &&
101 Morpheus::is_serial_execution_space_v<ExecSpace> &&
102 Morpheus::has_access_v<typename ExecSpace::execution_space, SourceType,
103 DestinationType>>::type* = nullptr) {
104 // Complexity: Linear. Specifically O(nnz(coo) + max(n_row,n_col))
105 using index_type = typename SourceType::index_type;
106
107 dst.resize(src.nrows(), src.ncols(), src.nnnz());
108
109 // compute number of non-zero entries per row of coo src
110 for (index_type n = 0; n < src.nnnz(); n++) {
111 dst.row_offsets(src.crow_indices(n))++;
112 }
113
114 // cumsum the nnz per row to get csr row_offsets
115 for (index_type i = 0, cumsum = 0; i < src.nrows(); i++) {
116 index_type temp = dst.row_offsets(i);
117 dst.row_offsets(i) = cumsum;
118 cumsum += temp;
119 }
120 dst.row_offsets(src.nrows()) = src.nnnz();
121
122 // write coo column indices and values into csr
123 for (index_type n = 0; n < src.nnnz(); n++) {
124 index_type row = src.crow_indices(n);
125 index_type dest = dst.row_offsets(row);
126
127 dst.column_indices(dest) = src.ccolumn_indices(n);
128 dst.values(dest) = src.cvalues(n);
129
130 dst.row_offsets(row)++;
131 }
132
133 for (index_type i = 0, last = 0; i <= src.nrows(); i++) {
134 index_type temp = dst.row_offsets(i);
135 dst.row_offsets(i) = last;
136 last = temp;
137 }
