245 inline void assign(const size_t n, const value_type val) {
246 using range_policy = Kokkos::RangePolicy<size_t, execution_space>;
247
248 /* Resize if necessary (behavior of std:vector) */
249 if (n > _size) {
250 this->resize(n);
251 }
252
253 range_policy policy(0, n);
254 Impl::set_functor f(_values, val);
255 Kokkos::parallel_for("Morpheus::DenseVector::assign", policy, f);
256 }
