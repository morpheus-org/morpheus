34template <typename ExecSpace, typename Vector>
35void inclusive_scan(
36 const Vector& in, Vector& out, typename Vector::index_type size,
37 typename Vector::index_type start,
38 typename std::enable_if_t<
39 Morpheus::is_dense_vector_format_container_v<Vector> &&
40 !Morpheus::is_generic_space_v<ExecSpace> &&
41 Morpheus::is_serial_execution_space_v<ExecSpace> &&
42 Morpheus::has_access_v<typename ExecSpace::execution_space, Vector>>* =
43 nullptr) {
44 using IndexType = typename Vector::index_type;
45
46 out[start] = in[start];
47 for (IndexType i = start + 1; i < start + size; i++) {
48 out[i] = out[i - 1] + in[i];
49 }
50}

73template <typename ExecSpace, typename Vector1, typename Vector2>
74void inclusive_scan_by_key(
75 const Vector1& keys, const Vector2& in, Vector2& out,
76 typename Vector2::index_type size, typename Vector2::index_type start,
77 typename std::enable_if_t<
78 Morpheus::is_dense_vector_format_container_v<Vector1> &&
79 Morpheus::is_dense_vector_format_container_v<Vector2> &&
80 !Morpheus::is_generic_space_v<ExecSpace> &&
81 Morpheus::is_serial_execution_space_v<ExecSpace> &&
82 Morpheus::has_access_v<typename ExecSpace::execution_space, Vector1,
83 Vector2>>* = nullptr) {
84 using IndexType = typename Vector2::index_type;
85 using KeyType = typename Vector1::value_type;
86 using ValueType = typename Vector2::value_type;
87
88 KeyType prev_key = keys[start];
89 ValueType prev_value = in[start];
90 out[start] = in[start];
91
92 for (IndexType i = start + 1; i < start + size; i++) {
93 KeyType key = keys[i];
94
95 if (prev_key == key)
96 out[i] = prev_value = prev_value + in[i];
97 else
98 out[i] = prev_value = in[i];
99
100 prev_key = key;
101 }
102}

104template <typename ExecSpace, typename Vector1, typename Vector2>
105void exclusive_scan_by_key(
106 const Vector1& keys, const Vector2& in, Vector2& out,
107 typename Vector2::index_type size, typename Vector2::index_type start,
108 typename std::enable_if_t<
109 Morpheus::is_dense_vector_format_container_v<Vector1> &&
110 Morpheus::is_dense_vector_format_container_v<Vector2> &&
111 !Morpheus::is_generic_space_v<ExecSpace> &&
112 Morpheus::is_serial_execution_space_v<ExecSpace> &&
113 Morpheus::has_access_v<typename ExecSpace::execution_space, Vector1,
114 Vector2>>* = nullptr) {
115 using IndexType = typename Vector2::index_type;
116 using KeyType = typename Vector1::value_type;
117 using ValueType = typename Vector2::value_type;
118
119 KeyType temp_key = keys[start];
120 ValueType temp_value = in[start];
121 ValueType next = ValueType(0);
122
123 // first one is init
124 out[start] = next;
125 next = next + temp_value;
126
127 for (IndexType i = start + 1; i < start + size; i++) {
128 KeyType key = keys[i];
129
130 // use temp to permit in-place scans
131 temp_value = in[i];
132
133 if (temp_key != key) next = ValueType(0); // reset sum
134
135 out[i] = next;
136 next = next + temp_value;
137
138 temp_key = key;
139 }
140}
