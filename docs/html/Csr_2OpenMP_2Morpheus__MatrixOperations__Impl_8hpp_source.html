38template <typename ExecSpace, typename Matrix, typename Vector>
39void update_diagonal(
40 Matrix& A, const Vector& diagonal,
41 typename std::enable_if_t<
42 Morpheus::is_csr_matrix_format_container_v<Matrix> &&
43 Morpheus::is_dense_vector_format_container_v<Vector> &&
44 !Morpheus::is_generic_space_v<ExecSpace> &&
45 Morpheus::is_openmp_execution_space_v<ExecSpace> &&
46 Morpheus::has_access_v<typename ExecSpace::execution_space, Matrix,
47 Vector>>* = nullptr) {
48 using index_type = typename Matrix::index_type;
49
50#pragma omp parallel for
51 for (index_type i = 0; i < A.nrows(); ++i) {
52 for (index_type jj = A.row_offsets(i); jj < A.row_offsets(i + 1); jj++) {
53 if (A.column_indices(jj) == i) {
54 A.values(jj) = diagonal[i];
55 break;
56 }
57 }
58 }
59}
