42class CudaWorkspace {
43 public:
44 CudaWorkspace() : _workspace(nullptr), _nbytes(0) {}
45
46 ~CudaWorkspace() { _free(); }
47
48 template <typename ValueType>
49 void allocate(size_t N) {
50 size_t bytes = N * sizeof(ValueType);
51 if ((bytes > _nbytes) && (_nbytes > 0)) {
52 _free();
53 }
54 _malloc(bytes);
55 }
56
57 template <typename ValueType>
58 ValueType* data() {
59 return reinterpret_cast<ValueType*>(_workspace);
60 }
61
62 private:
63 void* _workspace;
64 size_t _nbytes;
65
66 void _malloc(size_t bytes) {
67 if (_workspace == nullptr) {
68 _nbytes = bytes;
69 if (cudaSuccess != cudaMalloc(&_workspace, bytes)) {
70 std::cout << "Malloc Failed on Device" << std::endl;
71 exit(EXIT_FAILURE);
72 }
73 }
74 }
75
76 void _free() {
77 if (_workspace != nullptr) {
78 if (cudaSuccess == cudaFree(_workspace)) {
79 _workspace = nullptr;
80 _nbytes = 0;
81 }
82 }
83 }
84};

100 void init() {

103 printf("CUBLAS initialization failed\n");
104 exit(EXIT_FAILURE);
105 }
106 cublasSetPointerMode(_handle, CUBLAS_POINTER_MODE_DEVICE);
107 }
108 }

112 private:
