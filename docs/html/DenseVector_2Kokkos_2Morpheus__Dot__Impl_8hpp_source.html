34template <typename ExecSpace, typename Vector1, typename Vector2>
35inline typename Vector1::value_type dot(
36 typename Vector1::index_type n, const Vector1& x, const Vector2& y,
37 typename std::enable_if_t<
38 Morpheus::is_dense_vector_format_container_v<Vector1> &&
39 Morpheus::is_dense_vector_format_container_v<Vector2> &&
40 Morpheus::is_generic_space_v<ExecSpace> &&
41 Morpheus::has_access_v<typename ExecSpace::execution_space, Vector1,
42 Vector2>>* = nullptr) {
43 using execution_space = typename ExecSpace::execution_space;
44 using IndexType = Kokkos::IndexType<typename Vector1::index_type>;
45 using range_policy = Kokkos::RangePolicy<IndexType, execution_space>;
46 using ValueArray = typename Vector1::value_array_type;
47 using value_type = typename Vector1::non_const_value_type;
48 using index_type = typename Vector1::index_type;
49
50 const ValueArray x_view = x.const_view(), y_view = y.const_view();
51 range_policy policy(0, n);
52
53 value_type result = value_type(0);
54
55 if (y.data() == x.data()) {
56 Kokkos::parallel_reduce(
57 "dot_same_data", policy,
58 KOKKOS_LAMBDA(const index_type& i, value_type& lsum) {
59 lsum += x_view[i] * x_view[i];
60 },
61 result);
62 } else {
63 Kokkos::parallel_reduce(
64 "dot", policy,
65 KOKKOS_LAMBDA(const index_type& i, value_type& lsum) {
66 lsum += x_view[i] * y_view[i];
67 },
68 result);
69 }
70
71 return result;
72}
