38template <typename ExecSpace, typename SourceType, typename DestinationType>
39void convert(
40 const SourceType& src, DestinationType& dst,
41 typename std::enable_if<
42 Morpheus::is_dia_matrix_format_container_v<SourceType> &&
43 Morpheus::is_dia_matrix_format_container_v<DestinationType> &&
44 !Morpheus::is_generic_space_v<ExecSpace> &&

46 Morpheus::has_access_v<typename ExecSpace::execution_space, SourceType,

49
50 dst.resize(src.nrows(), src.ncols(), src.nnnz(),
51 src.cdiagonal_offsets().size());
52
53 // element-wise copy of indices and values
54 for (index_type n = 0; n < src.cdiagonal_offsets().size(); n++) {
55 dst.diagonal_offsets(n) = src.cdiagonal_offsets(n);
56 }
57
58 for (index_type j = 0; j < src.cvalues().ncols(); j++) {
59 for (index_type i = 0; i < src.cvalues().nrows(); i++) {
60 dst.values(i, j) = src.cvalues(i, j);
61 }

65template <typename ExecSpace, typename SourceType, typename DestinationType>
66void convert(
67 const SourceType& src, DestinationType& dst,
68 typename std::enable_if<
69 Morpheus::is_dia_matrix_format_container_v<SourceType> &&
70 Morpheus::is_coo_matrix_format_container_v<DestinationType> &&
71 !Morpheus::is_generic_space_v<ExecSpace> &&
72 Morpheus::is_serial_execution_space_v<ExecSpace> &&
73 Morpheus::has_access_v<typename ExecSpace::execution_space, SourceType,
74 DestinationType>>::type* = nullptr) {
75 using index_type = typename SourceType::index_type;
76 using value_type = typename SourceType::value_type;
77
78 dst.resize(src.nrows(), src.ncols(), src.nnnz());
79
80 const index_type ndiag = src.cvalues().ncols();
81
82 for (index_type i = 0, nnzid = 0; i < ndiag; i++) {
83 const index_type k = src.cdiagonal_offsets(i);
84
85 const index_type i_start = std::max<index_type>(0, -k);
86 const index_type j_start = std::max<index_type>(0, k);
87
88 // number of elements to process in this diagonal
89 const index_type N = std::min(src.nrows() - i_start, src.ncols() - j_start);
90
91 for (index_type n = 0; n < N; n++) {
92 const value_type temp = src.cvalues(i_start + n, i);
93
94 if (temp != value_type(0)) {
95 dst.row_indices(nnzid) = i_start + n;
96 dst.column_indices(nnzid) = j_start + n;
97 dst.values(nnzid) = temp;
98 nnzid = nnzid + 1;
99 }
100 }
101 }
102
103 if (!dst.is_sorted()) {
104 dst.sort_by_row_and_column();
105 }
106}

108template <typename ExecSpace, typename SourceType, typename DestinationType>
109void convert(
110 const SourceType& src, DestinationType& dst,
111 typename std::enable_if<
112 Morpheus::is_coo_matrix_format_container_v<SourceType> &&
113 Morpheus::is_dia_matrix_format_container_v<DestinationType> &&
114 !Morpheus::is_generic_space_v<ExecSpace> &&
115 Morpheus::is_serial_execution_space_v<ExecSpace> &&
116 Morpheus::has_access_v<typename ExecSpace::execution_space, SourceType,
117 DestinationType>>::type* = nullptr) {
118 using index_type = typename SourceType::index_type;
119
120 if (src.nnnz() == 0) {
121 dst.resize(src.nrows(), src.ncols(), src.nnnz(), 0);
122 return;
123 }
124
125 std::vector<index_type> diag_map(src.nnnz(), 0);
126
127 // Find on which diagonal each entry sits on
128 for (index_type n = 0; n < index_type(src.nnnz()); n++) {
129 diag_map[n] = src.ccolumn_indices(n) - src.crow_indices(n);
130 }
131
132 // Create unique diagonal set
133 std::set<index_type> diag_set(diag_map.begin(), diag_map.end());
134 index_type ndiags = index_type(diag_set.size());
135
136 if (dst.exceeds_tolerance(src.nrows(), src.nnnz(), ndiags)) {
137 throw Morpheus::FormatConversionException(
138 "DiaMatrix fill-in would exceed maximum tolerance");
139 }
140
141 dst.resize(src.nrows(), src.ncols(), src.nnnz(), ndiags);
142
143 for (auto it = diag_set.cbegin(); it != diag_set.cend(); ++it) {
144 auto i = std::distance(diag_set.cbegin(), it);
145 dst.diagonal_offsets(i) = *it;
146 }
147
148 for (index_type n = 0; n < index_type(src.nnnz()); n++) {
149 for (index_type i = 0; i < index_type(ndiags); i++) {
150 if (diag_map[n] == dst.diagonal_offsets(i)) {
151 dst.values(src.crow_indices(n), i) = src.cvalues(n);
152 break;
153 }
154 }
155 }
156}
