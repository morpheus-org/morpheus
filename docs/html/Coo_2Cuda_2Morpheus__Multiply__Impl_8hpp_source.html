67template <typename Matrix, typename Vector1, typename Vector2>
68void __spmv_coo_serial(const Matrix& A, const Vector1& x, Vector2& y,
69 const bool init) {
70 using index_type = typename Matrix::index_type;
71 using value_type = typename Matrix::value_type;
72 const index_type* I = A.crow_indices().data();
73 const index_type* J = A.ccolumn_indices().data();
74 const value_type* V = A.cvalues().data();
75
76 const value_type* x_ptr = x.data();
77 value_type* y_ptr = y.data();
78
79 if (init) {
80 y.assign(y.size(), 0);
81 }
82
83 Kernels::spmv_coo_serial_kernel<index_type, value_type>
84 <<<1, 1>>>(A.nnnz(), I, J, V, x_ptr, y_ptr);
85
86#if defined(DEBUG) || defined(MORPHEUS_DEBUG)
87 getLastCudaError("spmv_coo_serial_kernel: Kernel execution failed");
88#endif

92// COO SpMV kernel which flattens data irregularity (segmented reduction)
94// Copyright 2008-2014 NVIDIA Corporation
95// spmv_coo_flat
96// The input coo_matrix must be sorted by row. Columns within each row
97// may appear in any order and duplicate entries are also acceptable.
98// This sorted COO format is easily obtained by expanding the row pointer
99// of a CSR matrix (csr.Ap) into proper row indices and then copying
100// the arrays containing the CSR column indices (csr.Aj) and nonzero values
101// (csr.Ax) verbatim. A segmented reduction is used to compute the per-row
102// sums.
103//
104//
105template <typename Matrix, typename Vector1, typename Vector2>
106void __spmv_coo_flat(const Matrix& A, const Vector1& x, Vector2& y,
107 const bool init) {
108 using index_type = typename Matrix::index_type;
109 using value_type = typename Matrix::value_type;
110
111 if (init) {
112 y.assign(y.size(), 0);
113 }
114
115 if (A.nnnz() == 0) {
116 // empty matrix
117 return;
118 } else if (A.nnnz() < static_cast<index_type>(CUDA_WARP_SIZE)) {
119 // small matrix
120 Kernels::spmv_coo_serial_kernel<index_type, value_type><<<1, 1, 0>>>(
121 A.nnnz(), A.crow_indices().data(), A.ccolumn_indices().data(),
122 A.cvalues().data(), x.data(), y.data());
123#if defined(DEBUG) || defined(MORPHEUS_DEBUG)
124 getLastCudaError("spmv_coo_serial_kernel: Kernel execution failed");
125#endif
126 return;
127 }
128
129 const size_t BLOCK_SIZE = 256;
130 const size_t MAX_BLOCKS = max_active_blocks(
131 Kernels::spmv_coo_flat_kernel<index_type, value_type, BLOCK_SIZE>,
132 BLOCK_SIZE, (size_t)0);
133 const size_t WARPS_PER_BLOCK = BLOCK_SIZE / CUDA_WARP_SIZE;
134
135 const size_t num_units = A.nnnz() / CUDA_WARP_SIZE;
136 const size_t num_warps = std::min(num_units, WARPS_PER_BLOCK * MAX_BLOCKS);
137 const size_t num_blocks = DIVIDE_INTO(num_warps, WARPS_PER_BLOCK);
138 const size_t num_iters = DIVIDE_INTO(num_units, num_warps);
139
140 const index_type interval_size = CUDA_WARP_SIZE * num_iters;
141
142 const index_type tail =
143 num_units * CUDA_WARP_SIZE; // do the last few nonzeros separately (fewer
144 // than WARP_SIZE elements)
145
146 const unsigned int active_warps =
147 (interval_size == 0) ? 0 : DIVIDE_INTO(tail, interval_size);
148
149 typename Matrix::index_array_type temp_rows(active_warps, 0);
150 typename Matrix::value_array_type temp_vals(active_warps, 0);
151
152 Kernels::spmv_coo_flat_kernel<index_type, value_type, BLOCK_SIZE>
153 <<<num_blocks, BLOCK_SIZE, 0>>>(
154 tail, interval_size, A.crow_indices().data(),
155 A.ccolumn_indices().data(), A.cvalues().data(), x.data(), y.data(),
156 temp_rows.data(), temp_vals.data());
157#if defined(DEBUG) || defined(MORPHEUS_DEBUG)
158 getLastCudaError("spmv_coo_flat_kernel: Kernel execution failed");
159#endif
160
161 Kernels::spmv_coo_reduce_update_kernel<index_type, value_type, BLOCK_SIZE>
162 <<<1, BLOCK_SIZE, 0>>>(active_warps, temp_rows.data(), temp_vals.data(),
163 y.data());
164#if defined(DEBUG) || defined(MORPHEUS_DEBUG)
165 getLastCudaError("spmv_coo_reduce_kernel: Kernel execution failed");
166#endif
167
168 Kernels::spmv_coo_serial_kernel<index_type, value_type>
169 <<<1, 1, 0>>>(A.nnnz() - tail, A.crow_indices().data() + tail,
170 A.ccolumn_indices().data() + tail,
171 A.cvalues().data() + tail, x.data(), y.data());
172#if defined(DEBUG) || defined(MORPHEUS_DEBUG)
173 getLastCudaError("spmv_coo_serial_kernel: Kernel execution failed");
174#endif
175}
